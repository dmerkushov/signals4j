16  package ru.dmerkushov.signals4j;
17  
18  import java.util.Objects;
19  import java.util.Set;
20  
21  /**
22   *
23   * @author dmerkushov
24   */
25  public final class Signals4j {
26  
27  	private SignalQueueCheckRunnable slotR;
28  	private Thread slotT;
29  	private static Signals4j instance = null;
30  
31  	/**
32  	 * Get the singleton instance of the framework general class
33  	 *
34  	 * @return
35  	 */
36  	public synchronized static Signals4j getInstance () {
37  		if (instance == null) {
38  			instance = new Signals4j ();
39  			instance.slotT.start ();
40  		}
41  
42  		return instance;
43  	}
44  
45  	private Signals4j () {
46  		slotR = new SignalQueueCheckRunnable ();
47  		slotT = new Thread (slotR);
48  		slotT.setName ("SlotThread");
49  		slotT.setDaemon (true);
50  	}
51  
52  	/**
53  	 * Connect a signal class to a slot. When a signal of this class is emitted,
54  	 * the slot will be executed
55  	 *
56  	 * @param <S>
57  	 * @param signalClass
58  	 * @param slot
59  	 */
60  	public <S extends Signal> void connect (Class<S> signalClass, Slot<S> slot) {
61  		Objects.requireNonNull (signalClass, "signalClass");
62  		Objects.requireNonNull (slot, "slot");
63  
64  		SignalSlotMap.getInstance ().connect (signalClass, slot);
65  	}
66  
67  	/**
68  	 * Connect a signal to a slot based on the signal's class. Thus, all signals
69  	 * belonging to the same class as this one, when emitted, will make this
70  	 * slot execute
71  	 *
72  	 * @param <S>
73  	 * @param signal
74  	 * @param slot
75  	 */
76  	public <S extends Signal> void connect (S signal, Slot<S> slot) {
77  		Objects.requireNonNull (signal, "signal");
78  		Objects.requireNonNull (slot, "slot");
79  
80  		SignalSlotMap.getInstance ().connect (signal, slot);
81  	}
82  
83  	/**
84  	 * Disconnect a signal class from a slot
85  	 *
86  	 * @param <S>
87  	 * @param signalClass
88  	 * @param slot
89  	 */
90  	public <S extends Signal> void disconnect (Class<S> signalClass, Slot<S> slot) {
91  		Objects.requireNonNull (signalClass, "signalClass");
92  		Objects.requireNonNull (slot, "slot");
93  
94  		SignalSlotMap.getInstance ().disconnect (signalClass, slot);
95  	}
96  
97  	/**
98  	 * Disconnect the signal's class from a slot
99  	 *
100 	 * @param <S>
101 	 * @param signal
102 	 * @param slot
103 	 */
104 	public <S extends Signal> void disconnect (S signal, Slot<S> slot) {
105 		Objects.requireNonNull (signal, "signal");
106 		Objects.requireNonNull (slot, "slot");
107 
108 		SignalSlotMap.getInstance ().disconnect (signal, slot);
109 	}
110 
111 	/**
112 	 * Disconnect all slots from the signal's class
113 	 *
114 	 * @param <S>
115 	 * @param signal
116 	 */
117 	public <S extends Signal> void disconnectAll (S signal) {
118 		Objects.requireNonNull (signal, "signal");
119 
120 		SignalSlotMap.getInstance ().disconnectAll (signal);
121 	}
122 
123 	/**
124 	 * Disconnect all slots from the signal class
125 	 *
126 	 * @param <S>
127 	 * @param signalClass
128 	 */
129 	public <S extends Signal> void disconnectAll (Class<S> signalClass) {
130 		Objects.requireNonNull (signalClass, "signalClass");
131 
132 		SignalSlotMap.getInstance ().disconnectAll (signalClass);
133 	}
134 
135 	/**
136 	 * Get slots connected to a specified signal class
137 	 *
138 	 * @param signalClass
139 	 * @return
140 	 */
141 	public Set<Slot> getSlots (Class<? extends Signal> signalClass) {
142 		Objects.requireNonNull (signalClass, "signalClass");
143 
144 		return SignalSlotMap.getInstance ().getSlots (signalClass);
145 	}
146 
147 	/**
148 	 * Get slots connected to the signal's class
149 	 *
150 	 * @param signal
151 	 * @return
152 	 */
153 	public Set<Slot> getSlots (Signal signal) {
154 		Objects.requireNonNull (signal, "signal");
155 
156 		return SignalSlotMap.getInstance ().getSlots (signal);
157 	}
158 
159 	/**
160 	 * Get the number of slots currently connected to the given signal class
161 	 *
162 	 * @param signalClass
163 	 * @return
164 	 */
165 	public int getSlotCount (Class<? extends Signal> signalClass) {
166 		Objects.requireNonNull (signalClass, "signalClass");
167 
168 		return SignalSlotMap.getInstance ().getSlotCount (signalClass);
169 	}
170 
171 	/**
172 	 * Get the number of slots currently connected to the given signal's class
173 	 *
174 	 * @param signal
175 	 * @return
176 	 */
177 	public int getSlotCount (Signal signal) {
178 		Objects.requireNonNull (signal, "signal");
179 
180 		return SignalSlotMap.getInstance ().getSlotCount (signal);
181 	}
182 
183 }
